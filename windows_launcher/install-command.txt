# Soul OS One-Line Installer for Windows
# Copy and paste this command into PowerShell (Run as Administrator):

Set-ExecutionPolicy Bypass -Scope Process -Force; [System.Net.ServicePointManager]::SecurityProtocol = [System.Net.ServicePointManager]::SecurityProtocol -bor 3072; iex ((New-Object System.Net.WebClient).DownloadString('https://raw.githubusercontent.com/vikrant-project/soul-os/main/windows_launcher/Install-SoulOS.ps1'))
